// 건축한계선 계산 (세그먼트별 차등 적용)
function calculateSetbackLine() {
    const setbackCoords = [];
    
    // 각 세그먼트별 후퇴 거리 (m를 mm로 변환)
    const segmentSetbacks = [
        3000, // S1: 남측 10m 도로
        3000, // S2: 남동 모서리
        2000, // S3: 동측 6m 도로
        1000, // S4: 서측 녹지
        1000, // S5: 서측 녹지
        1000, // S6: 서측 녹지
        1000, // S7: 서측 녹지
        2500, // S8: 북측 10m 도로
        2500, // S9: 북동 모서리
        3000  // S10: 남측 10m 도로
    ];
    
    // 각 점에 대해 인접한 두 변의 수직선을 계산하여 교점 찾기
    for (let i = 0; i < boundaryCoords.length; i++) {
        const prev = (i - 1 + boundaryCoords.length) % boundaryCoords.length;
        const next = (i + 1) % boundaryCoords.length;
        
        // 현재 점에서 이전 점으로의 벡터
        const v1 = {
            x: boundaryCoords[prev].x - boundaryCoords[i].x,
            y: boundaryCoords[prev].y - boundaryCoords[i].y
        };
        
        // 현재 점에서 다음 점으로의 벡터
        const v2 = {
            x: boundaryCoords[next].x - boundaryCoords[i].x,
            y: boundaryCoords[next].y - boundaryCoords[i].y
        };
        
        // 각 벡터의 수직 벡터 (내부 방향)
        const n1 = normalize({ x: -v1.y, y: v1.x });
        const n2 = normalize({ x: v2.y, y: -v2.x });
        
        // 두 세그먼트의 후퇴거리
        const setback1 = segmentSetbacks[prev];
        const setback2 = segmentSetbacks[i];
        
        // 각 변에서 후퇴한 선의 교점 계산
        const p1 = {
            x: boundaryCoords[i].x + n1.x * setback1,
            y: boundaryCoords[i].y + n1.y * setback1
        };
        
        const p2 = {
            x: boundaryCoords[i].x + n2.x * setback2,
            y: boundaryCoords[i].y + n2.y * setback2
        };
        
        // 두 수직선의 교점을 구하는 더 정확한 방법
        const angle = Math.atan2(n2.y, n2.x) - Math.atan2(n1.y, n1.x);
        const avgSetback = (setback1 + setback2) / 2;
        const factor = avgSetback / Math.sin(Math.abs(angle) / 2);
        
        const bisector = normalize({
            x: n1.x + n2.x,
            y: n1.y + n2.y
        });
        
        setbackCoords.push({
            x: boundaryCoords[i].x + bisector.x * factor * 0.7,
            y: boundaryCoords[i].y + bisector.y * factor * 0.7
        });
    }
    
    return setbackCoords;
}

// 벡터 정규화 함수
function normalize(v) {
    const length = Math.sqrt(v.x * v.x + v.y * v.y);
    return {
        x: v.x / length,
        y: v.y / length
    };
}
